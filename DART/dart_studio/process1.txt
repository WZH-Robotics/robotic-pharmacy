## 주문 내용을 사용할 수 있게 슬라이싱 및 리스트로 변환
def order_msg_slicer(order_msg):
    # slicing data
    sliced_lists = order_msg.split(",") # 받은 메세지 쉼표 기준으로 slice
    # put sliced data into lists
    medicine, value = list(), list()
    
    for sliced_list in sliced_lists:
        tp_log("{0}".format(sliced_list))
        medicine.append(sliced_list.split(":")[0])
        value.append(sliced_list.split(":")[1])
    return medicine, value

def take_medicine(ser, arduino_num, value):
    # 위치에 따라 상대이동 방향 및 거리 지정
    if arduino_num<5:
        delta_distance = posx(-33,0,0,0,0,0) # 선반에 맞게 movel 좌표 선언
        home_pose = mid_pos1
    else:
        delta_distance = posx(0,33,0,0,0,0) # 선반에 맞게 movel 좌표 선언
        home_pose = mid_pos2
        
    # value를 아침 점심 저녁으로 나눈다.
    morning = int(value[0])
    afternoon = int(value[1])
    evening = int(value[2])
    
    # 알약 추출 및 이동
    send_signal_to_drop(ser, arduino_num, morning) # 아두이노로 추출 신호 송신
    wait(0.3)
    if (morning>0):
        tp_log("wait_for_signal(ser)") # 아두이노로부터 추출 완료 신호를 대기
        wait(6)
    movel(delta_distance, vel=1000, mod=DR_MV_MOD_REL)
    wait(0.3)
    
    send_signal_to_drop(ser, arduino_num, afternoon) # 아두이노로 추출 신호 송신
    wait(0.3)
    if (afternoon>0):
        tp_log("wait_for_signal(ser)") # 아두이노로부터 추출 완료 신호를 대기
        wait(6)
    movel(delta_distance, vel=1000, mod=DR_MV_MOD_REL)
    wait(0.3)
    
    send_signal_to_drop(ser, arduino_num, evening) # 아두이노로 추출 신호 송신
    wait(0.3)
    if (evening>0):
        tp_log("wait_for_signal(ser)") # 아두이노로부터 추출 완료 신호를 대기
        wait(6)
    movel(home_pose)
    
def send_signal_to_drop(ser, arduino_num, times):
    # times만큼 반복
    while times:
        ser_msg = "{0}".format(arduino_num) # "1" + " " + f"{arduino_num}" + "535"
        serial_write(ser, ser_msg.encode()) # 실제 추출 신호 송신
        wait(1)
        
        # 떨어졌으면 times를 1씩 뺀다.
        medi_dropped = True #get_data_from_arduino()
        if medi_dropped == True:
            tp_log("times-=1")
            wait(0.3)
            times-=1
